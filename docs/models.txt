USER ( PK=ID , EMAIL , NAME , PASSWORD )
COLLABORATION ( PK=ID , USER* , PUBLICATION* , ROLE )
PUBLICATION ( PK=ID , TITLE , DESCRIPTION , DATE , OWNER* ) 
// PUBLICATION.OWNER IN USER.ID
CONTENT ( PK = ID , TITLE , DESCRIPTION , TYPE , PUBLICATION* , CREATOR* , RESPONSIBLE* , CONTENT* , + OPTIONS )
// CONTENT.CONTENT IN TEXTCONTENT.ID OR MEDIACONTENT.ID
TEXTCONTENT ( PK = ID , DESCRIPTION , MAINCONTENT* , HTML )
// TEXTCONTENT.HTML USES CK-EDITOR
MEDIACONTENT ( PK = ID , DESCRIPTION , MAINCONTENT* , ROUTE )
// MEDIACONTENT.ROUTE CAN BE INTERNAL OR URL

content options are: 
- top: starting row
- left: starting column
- width: number of columns
- auto-height: boolean
- height: number of rows
- center-v: boolean
- center-h: boolean
